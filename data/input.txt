def binary_search(arr, target):
    low = 0
    high = len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

def calculate_gradient(x, y, weight, bias, learning_rate):
    prediction = (weight * x) + bias
    error = prediction - y
    d_weight = error * x
    d_bias = error
    new_weight = weight - (learning_rate * d_weight)
    new_bias = bias - (learning_rate * d_bias)
    return new_weight, new_bias

def get_api_status(url):
    import requests
    try:
        response = requests.get(url, timeout=5)
        return {"status": response.status_code, "online": True}
    except:
        return {"status": 500, "online": False}

def bubble_sort(data):
    n = len(data)
    for i in range(n):
        for j in range(0, n - i - 1):
            if data[j] > data[j + 1]:
                data[j], data[j + 1] = data[j + 1], data[j]
    return data

def recursive_backtrack(path, options):
    if not options:
        return path
    for choice in options:
        if choice not in path:
            path.append(choice)
            result = recursive_backtrack(path, [o for o in options if o != choice])
            if result: return result
            path.pop()
    return None

def factorial_logic(n):
    if n == 0 or n == 1:
        return 1
    return n * factorial_logic(n - 1)

def filter_neural_data(raw_string):
    import re
    clean = re.sub(r'[^a-zA-Z0-9\s]', '', raw_string)
    return clean.strip().lower()

def compute_moving_average(prices, window=3):
    averages = []
    for i in range(len(prices) - window + 1):
        segment = prices[i:i+window]
        averages.append(sum(segment) / window)
    return averages
